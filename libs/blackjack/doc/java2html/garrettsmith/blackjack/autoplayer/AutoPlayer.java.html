168     public
169     void solve()
170         throws IOException {
171         try {
172             _checkThread = new StopCheckThread( this, Thread.currentThread() );
173             synchronized ( _checkThread ) {
174                 _checkThread.start();
175                 while ( !_checkThread.isStarted() ) {
176                     _checkThread.wait();
177                 }
178             }
179             int i;
180             while ( this.shouldSolve() ) {
181                 _blackjack.playGame( _player, 1.0 );
182             }
183             _autoPlayer._player.signalDone();
184         }
185         catch ( InterruptedException iE ) {
186           // TODO use log4j, should System.exit?
187             iE.printStackTrace();
188             System.exit( 1 );
189         }

219         public
220         void run() {
221 
222             byte[] data = new byte[ 2048 ];
223             DatagramSocket socket = null;
224             DatagramPacket packet = null;
225 
226             synchronized ( this ) {
227 
228                 try {
229 
230                     socket = new DatagramSocket( _LISTEN_PORT );
231                     packet = new DatagramPacket( data, data.length );
232                     _started = true;
233                     this.notifyAll();
234                 }
235                 catch ( SocketException socketE ) {
236 
237                     // A SocketException will be thrown if port _LISTEN_PORT is
238                     // already being used on this host (or, less likely, for
239                     // other reasons).  In this case this object communicates
240                     // to the Solver by signaling it to stop and by
241                     // interrupting its thread, causing an InterruptedException
242                     // to be thrown.
243                     // See Solver.solve().
244                     System.out.println( "socket error: "
245                                                        + socketE.getMessage() );
246                     _solverThread.interrupt();
247                 }
248             }
249 
250             try {
251 
252                 while ( true ) {
253 
254                     socket.receive( packet );
255                     String message = new String( packet.getData(),
256                                                  packet.getOffset(),
257                                                  packet.getLength() );
258 
259                     if ( _STOP_MESSAGE.equals( message ) ) {
260 
261                         _autoPlayer.stop();
262                     }
263                 }
264             }
265             catch ( IOException ioE ) {
266 
267                 // In case of an exception being thrown in this try-catch block
268                 // this thread signals to the Solver via simply calling its stop
269                 // method.  No notification is needed, as the Solver is already
270                 // polling its shouldSolve() method, which checks to see if
271                 // stop() has been called.
272                 System.out.println( "IO error: " + ioE.getMessage() );
273                 _autoPlayer.stop();
274             }
275         }

303     private static class Options {
304         
305         private Options( String[] arguments ) {
306 
307             LongOpt[] longOpts = new LongOpt[ 7 ];
308 
309             // options
310             longOpts[ 0 ] = new LongOpt( "help",
311                                          LongOpt.NO_ARGUMENT,
312                                          null,
313                                          'h' );
314             longOpts[ 1 ] = new LongOpt( "num-threads",
315                                          LongOpt.REQUIRED_ARGUMENT,
316                                          null,
317                                          'n' );
318             longOpts[ 2 ] = new LongOpt( "port",
319                                          LongOpt.REQUIRED_ARGUMENT,
320                                          null,
321                                          'p' );
322             longOpts[ 3 ] = new LongOpt( "properties-file",
323                                          LongOpt.REQUIRED_ARGUMENT,
324                                          null,
325                                          'P' );
326 
327             // execution modes
328             longOpts[ 4 ] = new LongOpt( "run",
329                                          LongOpt.NO_ARGUMENT,
330                                          null,
331                                          'r' );
332             longOpts[ 5 ] = new LongOpt( "report",
333                                          LongOpt.OPTIONAL_ARGUMENT,
334                                          null,
335                                          'R' );
336             longOpts[ 6 ] = new LongOpt( "stop",
337                                          LongOpt.NO_ARGUMENT,
338                                          null,
339                                          's' );
340             Getopt getopt = new Getopt( this.getClass().getName(),
341                                         arguments,
342                                         ":hrsR::n:p:P:",
343                                         longOpts );
344 
345             int value;
346             while ( ( value = getopt.getopt() ) != -1 ) {
347 
348                 switch ( value ) {
349 
350                     case 'h':
351                         helpSelected = true;
352                         break;
353 
354                     case 'r':
355                         commandSpecified = true;
356                         command = Options.RUN;
357                         break;
358 
359                     case 's':
360                         commandSpecified = true;
361                         command = Options.STOP;
362                         break;
363                         
364                     case 'R':
365                         commandSpecified = true;
366                         command = Options.REPORT;
367                         reportType = ( getopt.getOptarg() == null
368                                        ? null
369                                        : getopt.getOptarg() );
370                         break;
371                         
372                     case 'n':
373                         try {
374                             
375                             numberOfThreads =
376                                          Integer.parseInt( getopt.getOptarg() );
377                         }
378                         catch ( NumberFormatException nfe ) {
379                          
380                             printBadNumberOfThreads( nfe.getMessage () );
381                             argumentsCorrect = false;
382                         }
383                         break;
384                     
385                     case 'p':
386                         try {
387                             
388                             portNumber = Integer.parseInt( getopt.getOptarg() );
389                         }
390                         catch ( NumberFormatException nfe ) {
391                          
392                             printBadPortNumber( nfe.getMessage () );
393                             argumentsCorrect = false;
394                         }
395                         break;
396                         
397                     case 'P':
398                         propertiesFile = new File( getopt.getOptarg() );
399                         break;
400                         
401                     case '?':
402                     case ':':
403                         argumentsCorrect = false;
404                         break;
405                
406                     default:
407                         throw new IllegalArgumentException(
408                             "getopt() returned '" + (char) value + "'" );
409                 }
410             }
411             // TODO further business validate options
412             // check port num positive int
413             // check thread num sane
414             // check report name sane
415             if ( !commandSpecified )
416                 argumentsCorrect = false;
417                 
418             if ( !argumentsCorrect || helpSelected )
419                 printHelp();
420             
421         } // private Options( String[] )
