122     public
123     String getCSVDataPoint() {
124 
125         // TODO potential bug: _SIX_DECIMALS_FORMAT may put commas in its output
126         return _numHandsPlayed + ","
127                  +_SIX_DECIMALS_FORMAT.format(getUnitsWonPerHandTimesThousand()) 
128                  + _SEPARATOR;
129     }

131     public
132     String getStatus() {
133 
134         StringBuffer result = new StringBuffer();
135         String stoppedStatus = "";
136         result.append( "Start Time:                " + _startDate.toString()
137                                                                  + _SEPARATOR );
138         if ( _stopDate != null ) {
139 
140             result.append( "Stop Time:                 " + _stopDate.toString()
141                                                                  + _SEPARATOR );
142         }
143         result.append( "Net winnings:              "
144             + _FORMAT.format( _netWinnings ) + _SEPARATOR
145             + "Number of hands played:    " + _FORMAT.format( _numHandsPlayed )
146             + _SEPARATOR + "Units won per hand X 1000: "
147             + _SIX_DECIMALS_FORMAT.format( getUnitsWonPerHandTimesThousand() )
148             + _SEPARATOR );
149         if ( _stopDate != null ) {
150 
151             double handsPerSecond = _numHandsPlayed * 1000 /
152                                  ( _stopDate.getTime() - _startDate.getTime() );
153             result.append( "Hands played per second:   "
154                               + _FORMAT.format( handsPerSecond ) + _SEPARATOR );
155         }
156         return result.toString();
157     }
158 
159     public
160     double getUnitsWonPerHandTimesThousand() {
161         
162         return _netWinnings * 1000 / _numHandsPlayed;
163     }
164     
165     public
166     void handFinished( Hand hand,
167                        double gainOrLoss,
168                        Result result,
169                        CardList dealerCards ) {
170 
171         _netWinnings += gainOrLoss;
172         _numHandsPlayed++;
173         if ( _numHandsPlayed % _writeThreshold == 0 ) {
174             
175             try {
176                 _output.write( getCSVDataPoint() );
177             }
178             catch ( IOException ioe ) {
179                 
180                 ioe.printStackTrace();
181                 // TODO shut down
182             }
183         }
184     }

186     public
187     void initialize( Properties properties )
188         throws Exception {
189 
190         String className = properties.getProperty( SELECTOR_CLASSNAME );
191         if ( className == null ) {
192 
193             throw new Exception( "property " + SELECTOR_CLASSNAME
194                                                                + " not found" );
195         }
196         _selector = ( StrategySelector )
197                                        Class.forName( className ).newInstance();
198         _selector.initialize( properties );
199         _writeThreshold = Long.parseLong(
200                                     properties.getProperty( WRITE_THRESHOLD ) );
201         _dataFile = new File( properties.getProperty( DATA_FILE_LOCATION ) );
202         if ( _dataFile.exists() )
203             _dataFile.delete();
204         _dataFile.createNewFile();
205         if ( !_dataFile.canWrite() )
206             throw new IOException( "cannot write to file "
207                                                 + _dataFile.getAbsolutePath() );
208         _output = new BufferedWriter( new FileWriter( _dataFile ) );
209         _startDate = new Date();
210     }
